const SHEET_NAME = 'Results'; const KEEP_MONTHS = 3; function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET_NAME); if (!sh) { sh = ss.insertSheet(SHEET_NAME); sh.appendRow(['Roll','Group','Institute','Year','GPA','TotalMarks','Status','SubjectData','SavedAt']); sh.setFrozenRows(1); sh.getRange(1,1,1,9).setFontWeight('bold').setBackground('#1C2B4A').setFontColor('#ffffff'); } return sh; } function doGet(e) { const p = e.parameter || {}; cleanOldRecords(); const rows = search(p.roll || '', p.group || '', p.year || ''); return ContentService .createTextOutput(JSON.stringify({ results: rows })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'save') saveRecords(data.records); } catch(err) {} return ContentService .createTextOutput(JSON.stringify({ ok: true })) .setMimeType(ContentService.MimeType.JSON); } function saveRecords(records) { const sh = getSheet(); cleanOldRecords(); records.forEach(rec => { const all = sh.getDataRange().getValues(); for (let i = all.length - 1; i >= 1; i--) { if (String(all[i][0]) === String(rec.roll) && all[i][1].toLowerCase() === rec.group.toLowerCase()) { sh.deleteRow(i + 1); } } sh.appendRow([ rec.roll, rec.group, rec.instituteName, rec.year, rec.finalGpa, rec.totalMarksObtained, rec.status, JSON.stringify(rec.subjectBreakdown), new Date().toISOString() ]); }); } function search(roll, group, year) { const sh = getSheet(); const data = sh.getDataRange().getValues(); const results = []; for (let i = 1; i < data.length; i++) { const r = data[i]; const rollMatch = !roll || String(r[0]).toLowerCase().includes(roll.toLowerCase()); const groupMatch = !group || r[1].toLowerCase() === group.toLowerCase(); const yearMatch = !year || r[3].toLowerCase() === year.toLowerCase(); if (rollMatch && groupMatch && yearMatch) { results.push({ roll: r[0], group: r[1], instituteName: r[2], year: r[3], finalGpa: r[4], totalMarksObtained: r[5], status: r[6], subjectBreakdown: JSON.parse(r[7] || '[]'), savedAt: r[8] }); } } return results; } function cleanOldRecords() { const sh = getSheet(); const cutoff = new Date(); cutoff.setMonth(cutoff.getMonth() - KEEP_MONTHS); const data = sh.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (new Date(data[i][8]) < cutoff) sh.deleteRow(i + 1); } }
