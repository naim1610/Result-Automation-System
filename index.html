🗂 Result Archive — Google Sheets

⚙️ One-time setup. Connect a Google Sheet as your result database. Takes about 3 minutes.
1
Create a new Google Sheet
Go to sheets.new and open a blank spreadsheet.
2
Open Apps Script
Click Extensions → Apps Script. Delete existing code, paste the code below:
// ============================================= // RAS — Google Apps Script Backend // Sheet 1: Results (final results archive) // Sheet 2: MarksSession (live marks progress) // ============================================= const SHEET_NAME = 'Results'; const SESSION_SHEET = 'MarksSession'; const KEEP_MONTHS = 3; // ---------- helpers ---------- function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET_NAME); if (!sh) { sh = ss.insertSheet(SHEET_NAME); sh.appendRow(['Roll','Group','Institute','Year','GPA','TotalMarks','Status','SubjectData','SavedAt']); sh.setFrozenRows(1); sh.getRange(1,1,1,9).setFontWeight('bold').setBackground('#1C2B4A').setFontColor('#ffffff'); } return sh; } function getSessionSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SESSION_SHEET); if (!sh) { sh = ss.insertSheet(SESSION_SHEET); // Columns: SessionKey | Config JSON | Students JSON | SavedAt sh.appendRow(['SessionKey','Config','StudentsData','SavedAt']); sh.setFrozenRows(1); sh.getRange(1,1,1,4).setFontWeight('bold').setBackground('#2C3E50').setFontColor('#ffffff'); } return sh; } // ---------- routing ---------- function doGet(e) { const p = e.parameter || {}; if (p.action === 'loadSession') return loadSession(p.key); cleanOldRecords(); const rows = search(p.roll || '', p.group || '', p.year || ''); return json({ results: rows }); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'save') saveRecords(data.records); if (data.action === 'saveSession') saveSession(data.key, data.config, data.students); } catch(err) { return json({ok:false,error:String(err)}); } return json({ ok: true }); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ---------- Results archive ---------- function saveRecords(records) { const sh = getSheet(); cleanOldRecords(); records.forEach(rec => { const all = sh.getDataRange().getValues(); for (let i = all.length - 1; i >= 1; i--) { if (String(all[i][0]) === String(rec.roll) && all[i][1].toLowerCase() === rec.group.toLowerCase()) { sh.deleteRow(i + 1); } } sh.appendRow([ rec.roll, rec.group, rec.instituteName, rec.year, rec.finalGpa, rec.totalMarksObtained, rec.status, JSON.stringify(rec.subjectBreakdown), new Date().toISOString() ]); }); } function search(roll, group, year) { const sh = getSheet(); const data = sh.getDataRange().getValues(); const results = []; for (let i = 1; i < data.length; i++) { const r = data[i]; const rollMatch = !roll || String(r[0]).trim().toLowerCase() === roll.trim().toLowerCase(); const groupMatch = !group || r[1].toLowerCase() === group.toLowerCase(); const yearMatch = !year || r[3].toLowerCase() === year.toLowerCase(); if (rollMatch && groupMatch && yearMatch) { results.push({ roll: r[0], group: r[1], instituteName: r[2], year: r[3], finalGpa: r[4], totalMarksObtained: r[5], status: r[6], subjectBreakdown: JSON.parse(r[7] || '[]'), savedAt: r[8] }); } } return results; } function cleanOldRecords() { const sh = getSheet(); const cutoff = new Date(); cutoff.setMonth(cutoff.getMonth() - KEEP_MONTHS); const data = sh.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (new Date(data[i][8]) < cutoff) sh.deleteRow(i + 1); } } // ---------- Marks session ---------- // Key = institute|group|year (uniquely identifies one batch) function saveSession(key, config, students) { const sh = getSessionSheet(); const data = sh.getDataRange().getValues(); // Delete existing row for this key for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === key) sh.deleteRow(i + 1); } sh.appendRow([key, JSON.stringify(config), JSON.stringify(students), new Date().toISOString()]); } function loadSession(key) { const sh = getSessionSheet(); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === key) { return json({ found: true, config: JSON.parse(data[i][1] || '{}'), students: JSON.parse(data[i][2] || '[]'), savedAt: data[i][3] }); } } return json({ found: false }); }
